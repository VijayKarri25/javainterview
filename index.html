// Used in: Logistics, Ride-sharing, Food Delivery, Fleet Management @RestController @RequestMapping("/api/v1/geo") public class GeoService { @PostMapping("/track") public ResponseEntity<GeoResponse> trackLocation( @Valid @RequestBody GeoRequest request) { // Validate coordinates if (request.lat() < -90 || request.lat() > 90) throw new InvalidCoordException("Latitude out of range"); // Calculate distance using Haversine formula double distance = haversine( prevLat, prevLng, request.lat(), request.lng()); // Check geofence GeofenceStatus status = geofenceService .check(request.lat(), request.lng()); // Save to time-series DB (InfluxDB/TimescaleDB) geoRepository.save(new GeoPoint( request.driverId(), request.lat(), request.lng(), request.timestamp())); return ResponseEntity.ok( new GeoResponse(distance, status)); } private double haversine(double lat1, double lon1, double lat2, double lon2) { double R = 6371; // Earth radius in km double dLat = Math.toRadians(lat2 - lat1); double dLon = Math.toRadians(lon2 - lon1); double a = Math.sin(dLat/2) * Math.sin(dLat/2) + Math.cos(Math.toRadians(lat1)) * Math.cos(Math.toRadians(lat2)) * Math.sin(dLon/2) * Math.sin(dLon/2); return R * 2 * Math.atan2( Math.sqrt(a), Math.sqrt(1-a)); } }
